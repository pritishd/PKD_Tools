71 # @return pynode of the skincluster that is made

73 def skinGeo(target,jointInfluences):
74  target = pm.PyNode(target)
75  jointInfluences = [pm.PyNode(inf) for inf in jointInfluences]
76 
77  #Apply Defomers
78  jnts = []
79  nonJnts = []
80 
81  for item in jointInfluences:
82  if item.getShape():
83  nonJnts.append(item)
84  else:
85  jnts.append(item)
86 
87  pm.select(target)
88  try:
89  mel.eval('doDetachSkin "2" { "1","1" };')
90  except:
91  pass
92 
93  #Skin to Joints
94  pm.select(jnts,target)
95  res = pm.skinCluster(tsb=1,mi=1)
96 
97  #Skin to non Joints
98  if nonJnts:
99  pm.select(nonJnts,target)
100  pm.skinCluster(res,e=1,ai=nonJnts)
101 
102  return res

107 # @param target (string/pynode) the geometery which contains the deformer
108 # @param deformer (string) the deformer which is queried
109 # @param multiple (bool) whether to query the first deformer found and return a list of deformer
110 # @return name / list of names of the target deformers
111 #
112 # @code
113 # import libUtilities
114 # #Query a single defomer
115 # libUtilities.get_target_defomer("myGeo","skinCluster")
116 # # Result: 'skinCluster1' #
117 # #Query multiple defomers
118 # libUtilities.get_target_defomer("myGeo","cluster",multiple=True)
119 # # Result: ['cluster1','cluster2','cluster3','cluster4']#
120 # @endcode
121 #
122 def get_target_defomer(target,deformer,multiple=False):
123  allDeformers = pm.listHistory(target,pdo=1)
124  if allDeformers:
125  allTargetDeformers = []
126  for dfrm in allDeformers :
127  if dfrm.type() == deformer:
128  if multiple:
129  allTargetDeformers.append(dfrm.name())
130  else:
131  return dfrm.name()
132  if allTargetDeformers:
133  allTargetDeformers.reverse()
134  return allTargetDeformers
