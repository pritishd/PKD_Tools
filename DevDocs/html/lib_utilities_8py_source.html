733 def freeze_transform(transform):
734  # Get the current lock status of the default attributes
735  defaultLockStatus = get_default_lock_status(transform)
736  transform = force_pynode(transform)
737  childrenLockStatus = {}
738  # Check to see if there are any children
739  if transform.getChildren(ad=1, type="transform"):
740  # Iterate through all the children
741  for childTransform in transform.getChildren(ad=1, type="transform"):
742  # Get the lock status of the children and store it in the dictonary
743  childrenLockStatus[childTransform] = get_default_lock_status(childTransform)
744  # Unlock the child transform status
745  unlock_default_attribute(childTransform)
746 
747  # Unlock default status
748  unlock_default_attribute(transform)
749 
750  # Freeze the tranform
751  pm.makeIdentity(transform, n=0, s=1, r=1, t=1, apply=True)
752 
753  # Reset the children lock status
754  for childTransform in childrenLockStatus:
755  set_lock_status(childTransform, childrenLockStatus[childTransform])
756  set_lock_status(transform, defaultLockStatus)
