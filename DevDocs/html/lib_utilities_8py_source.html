728 def freeze_transform(transform):
729  # Get the current lock status of the default attributes
730  defaultLockStatus = get_default_lock_status(transform)
731  transform = force_pynode(transform)
732  childrenLockStatus = {}
733  # Check to see if there are any children
734  if transform.getChildren(ad=1, type="transform"):
735  # Iterate through all the children
736  for childTransform in transform.getChildren(ad=1, type="transform"):
737  # Get the lock status of the children and store it in the dictonary
738  childrenLockStatus[childTransform] = get_default_lock_status(childTransform)
739  # Unlock the child transform status
740  unlock_default_attribute(childTransform)
741 
742  # Unlock default status
743  unlock_default_attribute(transform)
744 
745  # Freeze the tranform
746  pm.makeIdentity(transform, n=0, s=1, r=1, t=1, apply=True)
747 
748  # Reset the children lock status
749  for childTransform in childrenLockStatus:
750  set_lock_status(childTransform, childrenLockStatus[childTransform])
751  set_lock_status(transform, defaultLockStatus)
