714 # @param transform: The tranform node that is being evaluated
715 #
716 def freeze_transform(transform):
717  # Get the current lock status of the default attributes
718  defaultLockStatus = get_default_lock_status(transform)
719  transform = pm.PyNode(transform)
720  childrenLockStatus = {}
721  # Check to see if there are any children
722  if transform.getChildren(ad=1, type="transform"):
723  # Iterate through all the children
724  for childTransform in transform.getChildren(ad=1, type="transform"):
725  # Get the lock status of the children and store it in the dictonary
726  childrenLockStatus[childTransform] = get_default_lock_status(childTransform)
727  # Unlock the child transform status
728  unlock_default_attribute(childTransform)
729 
730  # Unlock default status
731  unlock_default_attribute(transform)
732 
733  # Freeze the tranform
734  pm.makeIdentity(transform, n=0, s=1, r=1, t=1, apply=True)
735 
736  # Reset the children lock status
737  for childTransform in childrenLockStatus:
738  set_lock_status(childTransform, childrenLockStatus[childTransform])
739  set_lock_status(transform, defaultLockStatus)
