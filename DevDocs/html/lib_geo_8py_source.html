65  geoWithHistory.append(meshParent)

68  # Check duplicate transform names

70  duplicateTransform.append(transform)

378 def convert_joint_to_cluster(targetGeo, skipList=[]):
379  # Convert to PyNode
380  targetGeo = libUtilities.force_pynode(targetGeo)
381  skin = libUtilities.get_target_defomer(targetGeo, "skinCluster")
382 
383  # Create the dictionary
384  clusterInfo = {}
385  # Progress Info
386  pm.progressWindow(title='Converting Skin To Cluster', progress=0, status='Progress: 0%')
387  totalJnts = len(skin.getInfluence()) - len(skipList)
388  currentJnt = 0.0
389 
390  # Go through Each Joint
391  for jnt in sorted(skin.getInfluence()):
392  if jnt.name() in skipList:
393  continue
394  # Get the vertex affected and the weight
395  vertZip, weightList = skin.getPointsAffectedByInfluence(jnt)
396  if not vertZip:
397  raise RuntimeError("Current Joint Has No Vertices:%s" % jnt)
398  pm.select(vertZip)
399 
400  # Iterate through selection and decompress vertic group into individual index
401  vertices = libUtilities.indexize_vertice_group(pm.selected())
402  # Select Vertices
403  libUtilities.select_vertices(targetGeo, vertices)
404  # Make a cluster
405  cltr, cTransform = pm.cluster(rel=1)
406  jntPos = pm.xform(jnt, q=1, ws=1, rp=1)
407  pm.setAttr(cTransform.scalePivot, jntPos)
408  pm.setAttr(cTransform.rotatePivot, jntPos)
409 
410  # Set the weight
411  for index, weight in zip(vertices, weightList):
412  cltr.weightList[0].weights[index].set(weight)
413 
414  # Add to dictionary
415  clusterInfo[jnt.name()] = {"cluster": cltr, "clusterHandle": cTransform}
416 
417  # Update Progress
418  currentJnt = currentJnt + 1.0
419  currentProgress = (currentJnt / totalJnts) * 100
420  pm.progressWindow(edit=True, progress=currentProgress, status=('Progress: ' + str(int(currentProgress)) + '%'))
421  pm.refresh()
422  pyLog.info("Converted: " + jnt.name())
423 
424  pm.progressWindow(endProgress=1)
425  return clusterInfo

440  source = str(args[0])

443  influenceShape = cmds.listRelatives(source, shapes=True, noIntermediate=True)[0]
444

448  exclusiveBind = kwargs.get('exclusiveBind', False)
449  autoWeightThreshold = kwargs.get('autoWeightThreshold', True)

454  wrapNode = wrapData[0]

457  cmds.setAttr(wrapNode + '.maxDistance', maxDistance)
458  cmds.setAttr(wrapNode + '.exclusiveBind', exclusiveBind)
459  cmds.setAttr(wrapNode + '.autoWeightThreshold', autoWeightThreshold)
460  cmds.setAttr(wrapNode + '.falloffMode', falloffMode)
461 
462  cmds.connectAttr(target + '.worldMatrix[0]', wrapNode + '.geomMatrix')
463 
464  # add influence
465  duplicateData = cmds.duplicate(source, name=source + 'Base')
466  base = duplicateData[0]
467  shapes = cmds.listRelatives(base, shapes=True)
468  baseShape = shapes[0]
469  cmds.hide(base)
470 
471  # create dropoff attr if it doesn't exist
472  if not cmds.attributeQuery('dropoff', n=source, exists=True):
473  cmds.addAttr(source, sn='dr', ln='dropoff', dv=4.0, min=0.0, max=20.0)

478  # create smoothness attr if it doesn't exist

482

484  if not cmds.attributeQuery('inflType', n=source, exists=True):
485  cmds.addAttr(source, at='short', sn='ift', ln='inflType', dv=2, min=1, max=2)
486 
487  cmds.connectAttr(influenceShape + '.worldMesh', wrapNode + '.driverPoints[0]')
488  cmds.connectAttr(baseShape + '.worldMesh', wrapNode + '.basePoints[0]')
489  cmds.connectAttr(source + '.inflType', wrapNode + '.inflType[0]')
490  cmds.connectAttr(source + '.smoothness', wrapNode + '.smoothness[0]')
491 
492  # if type nurbsCurve or nurbsSurface
493  if cmds.nodeType(influenceShape) == 'nurbsCurve' or cmds.nodeType(influenceShape) == 'nurbsSurface':
494  # create the wrapSamples attr if it doesn't exist
495  if not cmds.attributeQuery('wrapSamples', n=source, exists=True):
496  cmds.addAttr(source, at='short', sn='wsm', ln='wrapSamples', dv=10, min=1)
497  cmds.setAttr(source + '.wsm', k=True)

500  cmds.connectAttr(baseShape + '.ws', wrapNode + '.basePoints[0]')
501  cmds.connectAttr(source + '.wsm', wrapNode + '.nurbsSamples[0]')
502 
503  cmds.connectAttr(source + '.dropoff', wrapNode + '.dropoff[0]')
504 
505  return pm.PyNode(wrapNode)

516 def createFollicle(position, geo=None):
517  if geo.getShape().type() not in ["nurbsSurface", "mesh"]:
518  raise ValueError("Geometry must be mesh of nurbSurface")
519  else:
520  transform_node = pm.createNode("transform")
521  transform_node.translate.set(position)
522 
523  # make vector product nodes to get correct rotation of the transform node
524  vector_product = pm.createNode("vectorProduct")
525  vector_product.operation.set(4)
526  transform_node.worldMatrix >> vector_product.matrix
527  transform_node.rotatePivot >> vector_product.input1
528 
529  # connect the correct position to a closest point on surface node created
530  if geo.getShape().type() == "nurbsSurface":
531  closest_position = pm.createNode("closestPointOnSurface", n=(transform_node + "_CPOS"))
532  geo.ws >> closest_position.inputSurface
533  else:
534  closest_position = pm.createNode("closestPointOnMesh", n=(transform_node + "_CPOS"))
535  geo.outMesh >> closest_position.inMesh
536 
537  vector_product.output >> closest_position.inPosition
538 
539  # create a follicle node and connect it
540 
541  follicle = pm.createNode("follicle")
542  follicle_transform = follicle.getParent()
543 
544  follicle.outTranslate >> follicle_transform.translate
545  follicle.outRotate >> follicle_transform.rotate
546 
547  if geo.getShape().type() == "nurbsSurface":
548  geo.local >> follicle.inputSurface
549  else:
550  geo.outMesh >> follicle.inputMesh
551 
552  geo.worldMatrix >> follicle.inputWorldMatrix
553 
554  follicle.parameterU.set(closest_position.parameterU.get())
555  follicle.parameterV.set(closest_position.parameterV.get())
556 
557  # Delete nodes
558  pm.delete(transform_node)
559  pm.delete(closest_position)
560  return follicle_transform
