42 def _build_ik_(metaClass, solver, handleSuffix, startJointNumber, endJointNumber):
43  name = utils.nameMe(metaClass.side, metaClass.part, handleSuffix)
44  startJoint = metaClass.jointSystem.joints[startJointNumber].shortName()
45  endJoint = metaClass.jointSystem.joints[endJointNumber].shortName()
46  ikHandle = pm.ikHandle(name=name, sj=startJoint, ee=endJoint, sol=solver, sticky="sticky")[0]
47  ikHandleMeta = core.MovableSystem(ikHandle.name())
48  metaClass.transferPropertiesToChild(ikHandleMeta, handleSuffix[0].lower() + handleSuffix[1:])
49  ikHandleMeta.v = False
50  # IK Handle needs to be in it's own group in case the polevector is not set. Otherwise if you reparent it
51  # the polevector value changes in relation to the parent space
52  # Create the parent meta
53  ikHandleMeta.part = handleSuffix
54  ikHandleMeta.addParent(snap=False)
55  # Set the pivot to the endJoint
56  libUtilities.snap_pivot(ikHandleMeta.prnt.mNode, endJoint)
57 
58  return ikHandleMeta
