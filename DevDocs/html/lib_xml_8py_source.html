112 def _ConvertDictToXmlRecurse(parent, dictitem):
113  assert type(dictitem) is not type([])
114 
115  if isinstance(dictitem, dict):
116  for (tag, child) in dictitem.iteritems():
117  if str(tag) == '_text':
118  parent.text = str(child)
119  elif type(child) is type([]):
120  # iterate through the array and convert
121  for listchild in child:
122  elem = ElementTree.Element(tag)
123  parent.append(elem)
124  _ConvertDictToXmlRecurse(elem, listchild)
125  else:
126  elem = ElementTree.Element(tag)
127  parent.append(elem)
128  _ConvertDictToXmlRecurse(elem, child)
129  else:
130  parent.text = str(dictitem)

143 def _ConvertXmlToDictRecurse(node, dictclass):
144  nodedict = dictclass()
145 
146  if len(node.items()) > 0:
147  # if we have attributes, set them
148  nodedict.update(dict(node.items()))
149 
150  for child in node:
151  # recursively add the element's children
152  newitem = _ConvertXmlToDictRecurse(child, dictclass)
153  if type(newitem) == type(""):
154  # Convert to a float item if string is a number
155  if newitem.replace('e-', '', 1).replace("-", "", 1).replace('.', '', 1).isdigit():
156  newitem = float(newitem)
157  if nodedict.has_key(child.tag):
158  # found duplicate tag, force a list
159  if type(nodedict[child.tag]) is type([]):
160  # append to existing list
161  nodedict[child.tag].append(newitem)
162  else:
163  # convert to list
164  nodedict[child.tag] = [nodedict[child.tag], newitem]
165  else:
166  # only one, directly set the dictionary
167  nodedict[child.tag] = newitem
168 
169  if node.text is None:
170  text = ''
171  else:
172  text = node.text.strip()
173 
174  if len(nodedict) > 0:
175  # if we have a dictionary add the text as a dictionary value (if there is any)
176  if len(text) > 0:
177  nodedict['_text'] = text
178  else:
179  # if we don't have child nodes or attributes, just set the text
180  nodedict = text
181 
182  return nodedict
