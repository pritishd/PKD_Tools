313  def _calcPositionFallOff_(self, center, overRideJoints=None):
314 
315  if center < 1:
316  raise ValueError('Centre must be positive')
317  # Is there a overide of the joints
318  if overRideJoints:
319  joints = overRideJoints
320  else:
321  # Get the number of joints
322  joints = float(len(self.jointSystem))
323 
324  # Init the weight list
325  falloff = []
326  # Will the fall off be mirrored if it goes beyond the centre
327  mirrorFallOff = False
328  if center / joints <= 0.5:
329  center = (joints - center) + 1
330  mirrorFallOff = True
331  # Iterate thought all the joints
332  for i in range(1, int(joints) + 1):
333  val = round((center - (float(abs(center - i)))) / center, 3)
334 
335  falloff.append(round(val, 3))
336  if mirrorFallOff:
337  falloff.reverse()
338  return falloff

395  def calculateHeatPoints(self):
396  currentSkinMap = self.preNormalisedMap
397  # Iterate through all the weight maps for all the CV
398  for weights in currentSkinMap:
399  # Get the maximum value
400  maxWeight = max(weights)
401  # Index of this maximum value
402  index = weights.index(maxWeight)
403  # New weightList
404  newWeights = []
405  # Is weight above the threshold of .98
406  if maxWeight > .98:
407  # Set the weight to maximum of 1
408  weights[index] = 1.0
409  # Is the sum total = 1.0. If not reset other weights
410  if sum(weights) != 1.0:
411  newWeights = [0] * self.numHighLevelCtrls
412  else:
413  # What is the difference between the weights
414  diffWeight = 1.0 - maxWeight
415  # copy the weight
416  newWeights = list(weights)
417  # Set the maxWeight to zero in the new list
418  newWeights[index] = 0
419  # Redistribute the difference in the proporation of the other weights
420  # This way all the weight will add to 1
421  newWeights = libMath.calculate_proportions(newWeights, diffWeight)
422 
423  # Set the new weights
424  if newWeights:
425  for i in range(self.numHighLevelCtrls):
426  if i != index:
427  weights[i] = newWeights[i]
