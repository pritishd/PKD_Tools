315  def _check_scene_(self):
316  # Get the errors
317  topNode = libGeo.get_top_node()
318 
319  warnWindow = libPySide.QWarningBox()
320 
321  if not topNode:
322  warnWindow.setText("No top group found")
323  warnWindow.setWindowTitle("Missing Top Group")
324  warnWindow.exec_()
325  return
326  else:
327  # Check that there is only top node in the scene
328  if libGeo.multiple_top_nodes_exists():
329  warnWindow.setText("Multiple top groups found")
330  warnWindow.setWindowTitle("Multiple Top Groups")
331  detailed = ("There are multiple top groups in the scene. "
332  "The tool would perform better if you merge the groups. "
333  "Otherwise this tool would work on the first top group which is determined by Maya.")
334  warnWindow.setDetailedText(detailed)
335  warnWindow.exec_()
336  return
337 
338  errorInfo = libGeo.find_hierarchy_errors(topNode)
339 
340  if errorInfo["Namespace Transform"]:
341  warnWindow.setText("Namespace transform found")
342  warnWindow.setWindowTitle("Namespace Error")
343  detailed = "The following transforms have namespace in them\n\n"
344  for namespace in errorInfo["Namespace Transform"]:
345  detailed += "select -add %s;\n" % namespace.name()
346  warnWindow.setDetailedText(detailed)
347  warnWindow.exec_()
348  return
349 
350  if errorInfo["Duplicate Transform"]:
351  warnWindow.setText("Duplicate transform found")
352  warnWindow.setWindowTitle("Duplicate Transform Error")
353  detailed = "The following transforms are duplicated\n\n"
354  for duplicate in errorInfo["Duplicate Transform"]:
355  detailed += "select -add %s;\n" % duplicate.name()
356  warnWindow.setDetailedText(detailed)
357  warnWindow.exec_()
358  return
359 
360  if errorInfo["Duplicate Shapes"]:
361  warnWindow.setText("Duplicate Shapes")
362  warnWindow.setWindowTitle("Duplicate Shapes Error")
363  detailed = "The following shapes are duplicated however they seem to appear to come under different parent name\n\n"
364  for duplicate in errorInfo["Duplicate Shapes"]:
365  detailed += "select -add %s;\n" % duplicate.name()
366  detailed += '\n\nThis tool can attempt to fix the problem. Run the scene checker again to see if problem is resolved'
367  fixButton = libPySide.QtGui.QPushButton("Attempt Fix")
368  abortButton = libPySide.QtGui.QPushButton("Abort")
369  warnWindow.addButton(abortButton, libPySide.QtGui.QMessageBox.NoRole)
370  warnWindow.addButton(fixButton, libPySide.QtGui.QMessageBox.YesRole)
371  warnWindow.setDetailedText(detailed)
372  ret = warnWindow.exec_()
373  if ret:
374  libGeo.fix_duplicates_shapes(errorInfo["Duplicate Shapes"])
375  return
376 
377  if errorInfo["Incorrect Shape Names"]:
378  warnWindow.setText("Incorrect Shape Names")
379  warnWindow.setWindowTitle("Incorrect Shape Names Error")
380  detailed = '''The following names of the shapes do not follow maya's "[parentName]Shape naming convention"\n\n'''
381  for duplicate in errorInfo["Duplicate Shapes"]:
382  detailed += "select -add %s;\n" % duplicate.name()
383  detailed += '\n\nThis tool can attempt to fix the problem. Run the scene checker again to see if problem is resolved'
384  warnWindow.addButton("Abort", libPySide.QtGui.QMessageBox.NoRole)
385  warnWindow.addButton("Attempt Fix", libPySide.QtGui.QMessageBox.YesRole)
386  warnWindow.setDetailedText(detailed)
387  ret = warnWindow.exec_()
388  if ret:
389  libGeo.fix_duplicates_shapes(errorInfo["Incorrect Shape Names"])
390  return
391 
392  if errorInfo["History Geos"]:
393  warnWindow.setText("Geo with history")
394  warnWindow.setWindowTitle("Geo History Error")
395  detailed = '''The following object may have construction history on them. This may not be issue as sometime maya thinks that shader assignement is construction history. However be careful if you have any deformers such blendshapes, as these will be baked during export process\n\n'''
396  for historyGeo in errorInfo["History Geos"]:
397  detailed += "select -add %s;\n" % historyGeo.name()
398  warnWindow.setDetailedText(detailed)
