323  warnWindow = libPySide.QWarningBox()

337  warnWindow.setDetailedText(detailed)

341  errorInfo = libGeo.find_heirachy_errors(topNode)

344  if errorInfo["Namespace Transform"]:

346  warnWindow.setWindowTitle("Namespace Error")
347  detailed = "The following transforms have namespace in them\n\n"
