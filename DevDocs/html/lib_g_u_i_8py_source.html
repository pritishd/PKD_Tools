301  def _check_scene_(self):
302  # Get the errors
303  topNode = libGeo.get_top_node()
304 
305  warnWindow = libPySide.QWarningBox()
306 
307  if not topNode:
308  warnWindow.setText("No top group found")
309  warnWindow.setWindowTitle("Missing Top Group")
310  warnWindow.exec_()
311  return
312  else:
313  # Check that there is only top node in the scene
314  if libGeo.multiple_top_nodes_exists():
315  warnWindow.setText("Multiple top groups found")
316  warnWindow.setWindowTitle("Multiple Top Groups")
317  detailed = """There are multiple top groups in the scene. The tool would perform better if you merge the groups.
318 Otherwise this tool would work on the first top group which is determined by Maya."""
319  warnWindow.setDetailedText(detailed)
320  warnWindow.exec_()
321  return
322 
323  errorInfo = libGeo.find_heirachy_errors(topNode)
324  detailed_text = ""
325 
326  if errorInfo["Namespace Transform"]:
327  warnWindow.setText("Namespace transform found")
328  warnWindow.setWindowTitle("Namespace Error")
329  detailed = "The following transforms have namespace in them\n\n"
330  for namespace in errorInfo["Namespace Transform"]:
331  detailed += "select -add %s;\n" % namespace.name()
332  warnWindow.setDetailedText(detailed)
333  warnWindow.exec_()
334  return
335 
336  if errorInfo["Duplicate Transform"]:
337  warnWindow.setText("Duplicate transform found")
338  warnWindow.setWindowTitle("Duplicate Transform Error")
339  detailed = "The following transforms are duplicated\n\n"
340  for duplicate in errorInfo["Duplicate Transform"]:
341  detailed += "select -add %s;\n" % duplicate.name()
342  warnWindow.setDetailedText(detailed)
343  warnWindow.exec_()
344  return
345 
346  if errorInfo["Duplicate Shapes"]:
347  warnWindow.setText("Duplicate Shapes")
348  warnWindow.setWindowTitle("Duplicate Shapes Error")
349  detailed = "The following shapes are duplicated however they seem to appear to come under different parent name\n\n"
350  for duplicate in errorInfo["Duplicate Shapes"]:
351  detailed += "select -add %s;\n" % duplicate.name()
352  detailed += '\n\nThis tool can attempt to fix the problem. Run the scene checker again to see if problem is resolved'
353  fixButton = libPySide.QtGui.QPushButton("Attempt Fix")
354  okButton = libPySide.QtGui.QPushButton("Abort")
355  warnWindow.addButton(okButton, libPySide.QtGui.QMessageBox.NoRole)
356  warnWindow.addButton(fixButton, libPySide.QtGui.QMessageBox.YesRole)
357  warnWindow.setDetailedText(detailed)
358  ret = warnWindow.exec_()
359  if ret:
360  libGeo.fix_duplicates_shapes(errorInfo["Duplicate Shapes"])
361  return
362

364  warnWindow.setText("Geo with history")

366  detailed = "The following mesh has history on them\n\n"

368  detailed += "select -add %s;\n" % historyGeo.name()
369  warnWindow.setDetailedText(detailed)
370  warnWindow.exec_()
371  return
