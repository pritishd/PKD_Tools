48 # Import a file without any namespace or any top groups. Works on both .ma and .mb extensions
49 #
50 # @code
51 # import libFile
52 # libFile.importFile(r"c:\myFolder\test.ma")
53 # @endcode
54 #
55 # @attention Watch out for Fashing names with existing nodes.
56 #
57 def importFile(filePath):
58  extension = None
59  # Determine what kind of maya file are we dealing with
60  if has_extension(filePath, "mb"):
61  extension = "mayaBinary"
62  elif has_extension(filePath, "obj"):
63  pm.loadPlugin("objExport")
64  extension = "OBJ"
65  else:
66  extension = "mayaAscii"
67  # Make sure the filepath is maya compliant filepath
68  filePath = linux_path(filePath)
69 
70  # Import the file
71  evalLine = 'file -import -type "%s" -ra true -rpr "PKDTemp" -options "v=0" -pr -loadReferenceDepth "all" "%s";' % (
72  extension, filePath)
73 
74  # Print out the file debug statement
75  print(evalLine)
76  mel.eval(evalLine)
77 
78  # Remove the namespace from the nodes.
79  for node in pm.ls("PKDTemp*"):
80  node.unlock()
81  newName = str(node).replace("PKDTemp_", "")
82  node.rename(newName)
83  if ":" in node.name():
84  print ("Unable to rename: " + node)
