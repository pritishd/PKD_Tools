40 # Import a file without any namespace or any top groups. Works on both .ma and .mb extensions
41 #
42 # @code
43 # import libFile
44 # libFile.importFile(r"c:\myFolder\test.ma")
45 # @endcode
46 #
47 # @attention Watch out for clashing names with existing nodes.
48 #
49 def importFile(filePath):
50  extension = None
51  #Determine what kind of maya file are we dealing with
52  if filePath.split(".")[-1] == "mb":
53  extension = "mayaBinary"
54  else:
55  extension = "mayaAscii"
56  #Make sure the filepath is maya compliant filepath
57  filePath = linux_path(filePath)
58 
59  #Import the file
60  evalLine = 'file -import -type "%s" -ra true -rpr "PKDTemp" -options "v=0" -pr -loadReferenceDepth "all" "%s";'%(extension,filePath)
61 
62  #Print out the file debug statement
63  print evalLine
64  mel.eval(evalLine)
65 
66  #Remove the namespace from the nodes.
67  for node in pm.ls("PKDTemp*"):
68  node.unlock()
69  node.rename(str(node).replace("PKDTemp_",""))
